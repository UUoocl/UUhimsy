Sub AddNotesToAltText()

Dim oSlide As Slide
Dim oShape As Shape
Dim Notes As String

For Each oSlide In ActivePresentation.Slides
'Delete existing notes alt text

DeleteAgain:
For Each oShape In oSlide.Shapes
    'Debug.Print (oSlide.SlideIndex & ",  " & oSlide.SlideNumber & ",  " & oShape.Name)
    If oShape.Name = "altTextNotes" Then
        oShape.Delete
    End If
Next

'check that all the altTextNotes have been deleted
For Each oShape In oSlide.Shapes
    Debug.Print (oSlide.SlideIndex & ",  " & oSlide.SlideNumber & ",  " & oShape.Name)
    If oShape.Name = "altTextNotes" Then
        GoTo DeleteAgain
    End If
Next

'Get slide notes
For Each oShape In oSlide.NotesPage.Shapes
    If oShape.PlaceholderFormat.Type = ppPlaceholderBody Then
       If oShape.HasTextFrame Then
          If oShape.TextFrame.HasText Then
             Notes = oShape.TextFrame.TextRange.Text
             Debug.Print (Notes)
          End If
       End If
    End If
Next

If Len(Notes) > 0 Then
    'create new notes alt texts
    Set oShape = oSlide.Shapes.AddShape(Type:=msoShapeRectangle, _
        Left:=50, Top:=50, Width:=100, Height:=200)
    With oShape
        .Name = "altTextNotes"
        .Fill.Transparency = 1
        .Line.Visible = msoFalse
        .AlternativeText = "{NOTES:" & Notes & "NOTES:}"
    End With

    Notes = ""
End If

Next
End Sub
